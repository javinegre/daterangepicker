Preset ranges

<div id="dtp-2" class="btn btn-toolbar pull-right">
    <i class="icon-calendar icon-large"></i>
</div>

$('#dtp-2').daterangepicker(
    {
        ranges: {
            'Today': {type: 'enabled', dates: ['today', 'today']},
            'Last week': {type: 'enabled', dates: [Date.today().add({ days: -6 }), 'today']},
            'Last month': {type: 'enabled', dates: [Date.today().add({ months: -1 }), 'today']},
            'Last 3 months': {type: 'enabled', dates: [Date.today().add({ months: -3 }), 'today']},
            'Last 6 months': {type: 'enabled', dates: [Date.today().add({ months: -6 }), 'today']}
        }
    }
);

Callback

Initial date: Final date: 
<div id="dtp-3" class="btn btn-toolbar">
    <i class="icon-calendar icon-large"></i>
</div>
Initial date: <input id="input-1" type="text">
Final date: <input id="input-2" type="text">

$('#dtp-3').daterangepicker(
    {
        ranges: {
            'Last month': {type: 'enabled', dates: [Date.today().add({ months: -1 }), 'today']},
            'Last 6 months': {type: 'enabled', dates: [Date.today().add({ months: -6 }), 'today']},
            'Last year': {type: 'enabled', dates: [Date.today().add({ years: -1 }), 'today']}
        }
    },
    function (start, end) {
        $('#input-1').val(start.toString('yyyy-MM-dd'));
        $('#input-2').val(end.toString('yyyy-MM-dd'));
    }
);
